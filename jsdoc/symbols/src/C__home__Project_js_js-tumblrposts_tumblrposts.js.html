  8  /**
  9  * Create an instance of TumblrPosts Class
 10  * @class Class for dealing with Tumblr Posts through JSONP access
 11  * @param {Object} config Configuration definition
 12  * @example var posts = new TumblrPosts({ domain:"example.tumblr.com", maxNum:150 });
 13  */
 14 var TumblrPosts = Class.create();
 15 TumblrPosts.prototype = {
 16 	
 17 	/**
 18 	 * Name of complete event as string ( default "complete" )
 19 	 * @constant
 20 	 * @type String
 21 	 */
 22 	EVENT_COMPLETE:"complete",
 23 	
 24 	/**
 25 	 * Name of progress event as string ( default "progress" )
 26 	 * @constant
 27 	 * @type String
 28 	 */
 29 	EVENT_PROGRESS:"progress",
 30 	
 31 	config:{
 32 		domain: null,
 33 		maxNum:0
 34 	},
 35 	
 36 	api:"/api/read/json/",
 37 	postsTotal:0,
 38 	posts:[],
 39 	allTags:null,
 40 	tags:null,
 41 	offset:0,
 42 	num:50,
 43 	
 44 	/**
 45 	 * @constructor
 46 	 * @ignore
 47 	 */
 48 	initialize:function( config  ){
 49 		$.extend( this.config, config );
 50 		this.api = ( this.config.domain ) ? "http://" + this.config.domain + this.api : this.api
 51 	},
 52 	
 53 	/**
 54 	 * Start loading Tumblr API and reading the posts.
 55 	 * @param {Object} data Object passed by jQuery method. ( Normaly you don't need to pass )
 56 	 * @param {String} status Status name passed by jQuery method. ( Normaly you don't need to pass )
 57 	 */
 58 	run:function( data, status ){
 59 		var s = this,
 60 			abort = false;
 61 		
 62 		if( !!data ){
 63 			this.postsTotal = data["posts-total"];
 64 			this.offset += this.num;
 65 			this.allTags = this.allTags || "";
 66 			if( !data.posts.length ){
 67 				this.trigger( this.EVENT_COMPLETE );
 68 				return;
 69 			}
 70 			$.each( data.posts, function(i,p){
 71 				if( s.config.maxNum > 0 && s.posts.length >= s.config.maxNum ){
 72 					s.trigger( s.EVENT_COMPLETE );
 73 					abort = true;
 74 					return false;
 75 				}
 76 				s.posts.push( p );
 77 				if( !p.tags ) return;
 78 				s.allTags += "," + p.tags.join(",");
 79 			});
 80 			if( abort ){ return; }
 81 			this.trigger( this.EVENT_PROGRESS );
 82 		}
 83 		
 84 		$.ajax({
 85 			url: this.api,
 86 			data:{
 87 				start:this.offset,
 88 				num:this.num
 89 			},
 90 			dataType:"jsonp",
 91 			success: function( data, status ){
 92 				s.run( data, status );
 93 			}
 94 		});
 95 	},
 96 	
 97 	/**
 98 	 * Get list of the tags from all the posts.
 99 	 * @returns {Array} Tags Array with objects which holds tag name and counts.
100 	 */
101 	getTags: function(){
102 		var s = this,
103 			tag_data = {};
104 			
105 		if( this.tags === null ){
106 			this.allTags = (this.allTags.replace(/^,/,"")).split(",");
107 			$.each( this.allTags, function( i, t ){
108 				tag_data[t] = (tag_data[t] ) ? tag_data[t] + 1 : 1;
109 			});
110 			this.tags = [];
111 			$.each( tag_data, function(k,n){
112 				s.tags.push({
113 					name:k,
114 					count:n
115 				});
116 			});
117 			this.tags.sort( function(a,b){
118 				return b.count - a.count;
119 			});
120 		}
121 		return this.tags;
122 	},
123 	
124 	/**
125 	 * Method to get list of the posts, you may use this when making the updated list.
126 	 * @param {Number} offset Number to start with. 
127 	 * @param {Number} count Count of posts you want to get.
128 	 * @return {Array} posts The list of posts.
129 	 */
130 	getPosts: function( offset, count ){
131 		var offset = offset || 0,
132 			count = count || 10,
133 			result = [];
134 		
135 		$.each( this.posts, function(i,p){
136 			if( i < offset ){ return; }
137 			if( i > offset + count ){ return false; }
138 			result.push( p );
139 		});
140 		return result;
141 	},
142 	
143 	/**
144 	 * Get title from the post object.
145 	 * @param {Object} post Tumblr Post Object
146 	 * @param {Number} count Character count you want to extract from the title strings.
147 	 * @return {String} title Title of Post
148 	 */
149 	getTitleByPost: function( post, count ){
150 		var title = post["regular-title"]
151 			|| post["photo-caption"]
152 			|| post["regular-body"]
153 			|| post["tags"][0]
154 			|| "" ;
155 		
156 		title = title.replace( /\<.+?\>/gi, "" );
157 		title = ( count ) ? title.substr( 0, count ) : title;
158 		return title;
159 	},
160 	
161 	/**
162 	 * Get the rate how does the loading API progress.
163 	 * 	You may use this when progress event fired.
164 	 * @return {Number} number Integer of the rate.
165 	 */
166 	getLoadedRate:function(){
167 		return Math.floor( 100 * this.posts.length / 
168 			Math.min(this.postsTotal, (this.config.maxNum||this.postsTotal) ) );
169 	}
170 };
